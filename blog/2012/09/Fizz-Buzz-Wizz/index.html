Fizz Buzz Wizz

23/09/2012 21:44

Kata completed in C# through TDD with NUnit.

Specification

For any given number, n, output Fizz, Buzz or Wizz (stacking) such that:

if n is a multiple of:3510
then output:FizzBuzzWizz

Examples:

  • 3 → Fizz
  • 10 → BuzzWizz
  • 15 → FizzBuzz
  • 50 → FizzBuzzWizz
using NUnit.Framework;

namespace FizzBuzzWizz
{
	[TestFixture]
	internal class FizzBuzzWizz
	{
		private static string GetFizzBuzzWizz(int i)
		{
			return GetFizz(i)
				   + GetBuzz(i)
				   + GetWizz(i);
		}

		private static string GetWizz(int i)
		{
			return IfDivisibleReturnString(i, 10, "Wizz");
		}


		private static string GetBuzz(int i)
		{
			return IfDivisibleReturnString(i, 5, "Buzz");
		}

		private static string GetFizz(int i)
		{
			return IfDivisibleReturnString(i, 3, "Fizz");
		}

		private static string IfDivisibleReturnString(int dividend, int divisor, string s)
		{
			return dividend%divisor == 0 ? s : "";
		}

		[Test]
		public void Fifteen_should_output_FizzBuzz()
		{
			string output = GetFizzBuzzWizz(15);
			Assert.That(output, Is.EqualTo("FizzBuzz"));
		}

		[Test]
		public void Fifty_should_output_BuzzWizz()
		{
			string output = GetFizzBuzzWizz(50);
			Assert.That(output, Is.EqualTo("BuzzWizz"));
		}

		[Test]
		public void Five_should_output_Buzz()
		{
			string output = GetFizzBuzzWizz(5);
			Assert.That(output, Is.EqualTo("Buzz"));
		}

		[Test]
		public void Six_should_output_Fizz()
		{
			string output = GetFizzBuzzWizz(6);
			Assert.That(output, Is.EqualTo("Fizz"));
		}

		[Test]
		public void Ten_should_output_Buzzwhiz()
		{
			string output = GetFizzBuzzWizz(10);
			Assert.That(output, Is.EqualTo("Buzzwhiz"));
		}

		[Test]
		public void Thirty_should_output_FizzBuzzwhiz()
		{
			string output = GetFizzBuzzWizz(30);
			Assert.That(output, Is.EqualTo("FizzBuzzwhiz"));
		}

		[Test]
		public void Three_should_output_Fizz()
		{
			string output = GetFizzBuzzWizz(3);
			Assert.That(output, Is.EqualTo("Fizz"));
		}
	}
}
